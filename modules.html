모듈 (Modules)

TypeScript와 Python은 모두 모듈화된 코드 구성을 지원합니다. 코드를 여러 파일로 나누고 필요한 기능을 가져와 사용하는 방법을 알아봅시다.

Python vs TypeScript 비교

Python

# 모듈 정의 (math_utils.py)
def add(a, b):
    return a + b

def subtract(a, b):
    return a - b

# 특정 함수만 export
__all__ = ['add', 'subtract']

# 다른 파일에서 모듈 가져오기 (main.py)
# 방법 1: 모듈 전체 가져오기
import math_utils
result = math_utils.add(5, 3)

# 방법 2: 특정 함수만 가져오기
from math_utils import add, subtract
result = add(5, 3)

# 방법 3: 모듈을 다른 이름으로 가져오기
import math_utils as mu
result = mu.add(5, 3)

TypeScript

// 모듈 정의 (mathUtils.ts)
export function add(a: number, b: number): number {
    return a + b;
}

export function subtract(a: number, b: number): number {
    return a - b;
}

// 기본 내보내기 (선택 사항)
export default {
    add,
    subtract
};

// 다른 파일에서 모듈 가져오기 (main.ts)
// 방법 1: 특정 함수만 가져오기
import { add, subtract } from './mathUtils';
const result = add(5, 3);

// 방법 2: 모듈 전체를 가져와 이름 붙이기
import * as mathUtils from './mathUtils';
const result2 = mathUtils.add(5, 3);

// 방법 3: 기본 내보내기 가져오기
import mathFunctions from './mathUtils';
const result3 = mathFunctions.add(5, 3);
