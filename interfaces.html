인터페이스 (Interfaces)

TypeScript의 인터페이스는 객체의 형태를 정의하는 강력한 방법입니다. Python에는 직접적인 대응 개념이 없지만, 타입 힌팅과 비슷한 목적으로 사용할 수 있습니다.

Python vs TypeScript 비교

Python

# Python에서는 타입 힌팅을 사용하여 비슷한 효과를 낼 수 있음
from typing import TypedDict, Protocol

# TypedDict를 사용한 방법
class User(TypedDict):
    name: str
    age: int
    email: str

# 객체 생성 (타입 검사는 IDE나 mypy 같은 도구로 수행)
user: User = {
    "name": "John",
    "age": 30,
    "email": "john@example.com"
}

# Protocol을 사용한 방법
class Printable(Protocol):
    def print_info(self) -> None: ...

class Report:
    def print_info(self) -> None:
        print("Report printed")

# Report 클래스는 Printable 프로토콜을 "구현"함

TypeScript

// TypeScript 인터페이스 정의
interface User {
    name: string;
    age: number;
    email: string;
}

// 인터페이스에 맞는 객체 생성
const user: User = {
    name: "John",
    age: 30,
    email: "john@example.com"
};

// 인터페이스로 메서드 시그니처 정의
interface Printable {
    printInfo(): void;
}

// 인터페이스 구현
class Report implements Printable {
    printInfo(): void {
        console.log("Report printed");
    }
}
